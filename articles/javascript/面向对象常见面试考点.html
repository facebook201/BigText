# 1 以下代码的输出结果

let obj = {
    2: 3,
    3: 4,
    length: 2,
    push: Array.prototype.push
}
/**
 * Array.prototype.push = function () {
        this[this.length] = val;
        this.length++;
    }

    obj.push(1)执行过程中
        obj[obj.length]=1 =>obj[2]=1
        obj.length++ =>obj.length=3
     obj.push(2)执行过程中
        obj[obj.length]=2 =>obj[3]=2
        obj.length++ =>obj.length=4
 */
obj.push(1);
obj.push(2);
console.log(obj);
//所以最终输出
// {
//     2: 1,
//     3: 2,
//     length: 4,
//     push: Array.prototype.push
// }
